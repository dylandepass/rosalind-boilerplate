## Theme
<ColorPalette>
  {Object.entries(theme.theme.color).map(([k, v]) => {
        if (
          typeof v === 'string' &&
          (v.match(/^#/) || v.match(/^rgb/) || k.match(/color/i))
        ) {
          return <ColorItem key={k} title={k} colors={{ [k]: v }} />;
        } else if (typeof v === 'object') {
          return (
            <GroupColorItem title={k} tokenGroup={v} tokenName={`--ros-theme-color-${k}-{level}`}/>
          );
        }
        return null;
      })}
</ColorPalette>
